# ════════════════════════════════════════════════════════════════════
# RAG API with vLLM - Quick Start Guide
# ════════════════════════════════════════════════════════════════════

# ─────────────────────────────────────────────────────────────────────
# Terminal 1 - vLLM Server
# ─────────────────────────────────────────────────────────────────────
cd /home/abali/GithubCode/vLLM_rag
source vllm-env/bin/activate

vllm serve Qwen/Qwen2.5-3B-Instruct \
    --port 8080 \
    --gpu-memory-utilization 0.85

# Bekle: "Uvicorn running on http://0.0.0.0:8080"

# ─────────────────────────────────────────────────────────────────────
# Terminal 2 - RAG API Server
# ─────────────────────────────────────────────────────────────────────
cd /home/abali/GithubCode/vLLM_rag
source vllm-env/bin/activate

python api_server.py

# Bekle: "Uvicorn running on http://0.0.0.0:8000"

# ─────────────────────────────────────────────────────────────────────
# Terminal 3 - Test & Benchmark
# ─────────────────────────────────────────────────────────────────────
cd /home/abali/GithubCode/vLLM_rag
source vllm-env/bin/activate

# 1. Health check
curl http://localhost:8000/health

# 2. Tek sorgu testi
curl -X POST http://localhost:8000/query \
  -H "Content-Type: application/json" \
  -d '{"question": "Yazilim muhendisligi nedir?"}'

# 3. Benchmark testleri (benchmarks klasöründen)
cd benchmarks

# Hızlı concurrent test
python concurrent_test.py

# Detaylı GPU benchmark
python benchmark.py --test-type load

# Sonuçları görselleştir
python visualize_results.py

# ─────────────────────────────────────────────────────────────────────
# VEYA Makefile ile (daha kolay!)
# ─────────────────────────────────────────────────────────────────────
cd /home/abali/GithubCode/vLLM_rag

make vllm          # Terminal 1: vLLM server başlat
make api           # Terminal 2: RAG API başlat
make test          # Terminal 3: Hızlı test
make benchmark     # Terminal 3: Detaylı benchmark
make visualize     # Terminal 3: Sonuçları görselleştir
make all-tests     # Terminal 3: Tüm testleri çalıştır
